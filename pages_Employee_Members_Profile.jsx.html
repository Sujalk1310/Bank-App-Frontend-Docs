/**
 * Member Profile page component (Employee view).
 * Displays and allows editing of a member's comprehensive profile information.
 * 
 * Features:
 * - Fetches member data based on `mid` route parameter.
 * - Fetches dropdown options (branches, titles, genders, occupations).
 * - Displays member details (name, ID, F/H name, DOB, age, income, gender, etc.).
 * - Handles KYC documents: Photo, Signatures (Applicant & Introducer), Address Proof, Aadhar, PAN.
 *   - Includes file upload inputs with validation (type, size).
 *   - Allows viewing uploaded PDFs.
 * - Manages Present and Permanent Addresses (with "Same as Present" toggle).
 * - Captures Contact Info (Email, Phone) with validation.
 * - Records Company/Occupation details.
 * - Manages Nominee details.
 * - Records Introducer details.
 * - Provides an "Edit" mode toggle (`editing`).
 * - Submits updated profile data (including files) via multipart/form-data API call.
 * 
 * @returns {JSX.Element} The member profile page component.
 */
const Profile = () => {
    const [memberName, setMemberName] = useState("");
    const [editing, setEditing] = useState(false);
    const [memberId, setMemberId] = useState('');
    const [fhName, setFhName] = useState('');
    const [sender, setSender] = useState(false);
    const [branches, setBranches] = useState([]);
    const [titles, setTitles] = useState([]);
    const [error, setError] = useState(false);
    const [selectedBranch, setSelectedBranch] = useState(null);
    const [selectedTitle, setSelectedTitle] = useState(null);
    const [motherName, setMotherName] = useState('');
    const [dob, setDob] = useState(new Date());
    const [age, setAge] = useState('');
    const [monthlyIncome, setMonthlyIncome] = useState('');
    const [formattedMonthlyIncome, setFormattedMonthlyIncome] = useState('');
    const [genderList, setGenderList] = useState([]);
    const [selectedGender, setSelectedGender] = useState(null);
    const [occupationList, setOccupationList] = useState([]);
    const [selectedOccupation, setSelectedOccupation] = useState(null);
    const [panNo, setPanNo] = useState('');
    const [correctPanNo, setCorrectPanNo] = useState(true);
    const [aadharNo, setAadharNo] = useState('');
    const [correctAadharNo, setCorrectAadharNo] = useState(true);
    const [isMarried, setIsMarried] = useState(false);
    const [ownHouseProperty, setOwnHouseProperty] = useState(false);
    const [selectedFhRelation, setSelectedFhRelation] = useState(null);
    const [signatureApplicant, setSignatureApplicant] = useState(null);
    const [signatureApplicantPreview, setSignatureApplicantPreview] = useState(null);
    const [signatureIntroducer, setSignatureIntroducer] = useState(null);
    const [signatureIntroducerPreview, setSignatureIntroducerPreview] = useState(null);
    const [photo, setPhoto] = useState(null);
    const [photoPreview, setPhotoPreview] = useState(null);
    const [addressProof, setAddressProof] = useState(null);
    const [addressProofType, setAddressProofType] = useState(null);
    const [aadharCard, setAadharCard] = useState(null);
    const [panCard, setPanCard] = useState(null);
    const [sameAsPresent, setSameAsPresent] = useState(false);
    const [email, setEmail] = useState('');
    const [correctEmail, setCorrectEmail] = useState(true);
    const [phoneNumber, setPhoneNumber] = useState('');
    const [correctPhoneNumber, setCorrectPhoneNumber] = useState(true);
    const [permanentAddress, setPermanentAddress] = useState('');
    const [presentAddress, setPresentAddress] = useState('');
    const [companyName, setCompanyName] = useState('');
    const [companyAddress, setCompanyAddress] = useState('');
    const [nomineeName, setNomineeName] = useState('');
    const [nomineeRelationship, setNomineeRelationship] = useState('');
    const [nomineeAge, setNomineeAge] = useState('');
    const [nomineePhone, setNomineePhone] = useState('');
    const [nomineeEmail, setNomineeEmail] = useState('');
    const [introducersName, setIntroducersName] = useState('');
    const [correctNomineeEmail, setCorrectNomineeEmail] = useState(true);
    const [correctNomineePhone, setCorrectNomineePhoneNumber] = useState(true);
    const [submitting, setSubmitting] = useState(false);

    const { mid } = useParams();
    const navigate = useNavigate();

    const PHOTO_MAX_FILE_SIZE = (import.meta.env.VITE_PHOTO_SIZE_LIMIT
        ? import.meta.env.VITE_PHOTO_SIZE_LIMIT * 1024 * 1024
        : 2 * 1024 * 1024); // Default 2MB if env is not set\

    const PDF_MAX_FILE_SIZE = (import.meta.env.VITE_PDF_SIZE_LIMIT
        ? import.meta.env.VITE_PDF_SIZE_LIMIT * 1024 * 1024
        : 2 * 1024 * 1024); // Default 2MB if env is not set\

    const fhRelation = [{
        label: "Father",
        value: 0,
    }, {
        label: "Husband",
        value: 1,
    }]

    const addressProofDocs = [{
        label: "Electricity Bill",
        value: 0
    }, {
        label: "Water Bill",
        value: 1,
    }]

    const dobChange = (date) => {
        const birthDate = new Date(date);
        const today = new Date();

        if (birthDate > today) {
            toast("DOB cannot be a future date!", {
                icon: "⚠️"
            });
            return;
        }
    
        let calculatedAge = today.getFullYear() - birthDate.getFullYear();
        const monthDiff = today.getMonth() - birthDate.getMonth();
        
        // If birthday hasn't happened this year, subtract one year
        if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {
            calculatedAge--;
        }
    
        setDob(date);
        setAge(calculatedAge);
    };

    const openFile = (pdfFile) => {
        if (!pdfFile) return;
        console.log(pdfFile);
    
        let fileURL;
        
        if (pdfFile instanceof Blob || pdfFile instanceof File) {
            fileURL = URL.createObjectURL(pdfFile);
        } else if (typeof pdfFile === "string") {
            fileURL = pdfFile; // It's already a URL
        } else {
            console.error("Invalid PDF file type");
            return;
        }
    
        window.open(fileURL, "_blank");
    
        // Revoke the object URL after opening to prevent memory leaks
        if (pdfFile instanceof Blob || pdfFile instanceof File) {
            setTimeout(() => URL.revokeObjectURL(fileURL), 1000);
        }
    };

    const getBranchList = async () => {
        try {
            const response = await getAPI("/branchmaster/get-all");
            const formattedBranches = response.branches.map((branch) => ({
                label: branch.branchName,
                value: branch._id,
            }));
            setBranches(formattedBranches);
            setSelectedBranch(formattedBranches.length ? formattedBranches[0] : null); // Ensure it's correctly formatted
        } catch (error) {
            setError(true);
            toast.error(error.message);
        } 
    };

    const getTitleList = async () => {
        try {
            const response = await getAPI("/title/get-titles");
            const formattedTitles = response.titles.map((title) => ({
                label: title.salutation,
                value: title._id,
            }));

            setTitles(formattedTitles);
            setSelectedTitle(formattedTitles.length ? formattedTitles[0] : null); // Ensure it's correctly formatted
        } catch (error) {
            setError(true);
            toast.error(error.message);
        }
    };

    const getGenderList = async () => {
        try {
            const response = await getAPI("/gender/get-genders");
            const formattedGenders = response.genderMasters.map((gender) => ({
                label: gender.genderType,
                value: gender._id,
            }));

            setGenderList(formattedGenders);
            setSelectedGender(formattedGenders.length ? formattedGenders[0] : null); // Ensure it's correctly formatted
        } catch (error) {
            setError(true);
            toast.error(error.message);
        }
    }

    const getOccupationList = async () => {
        try {
            const response = await getAPI("/occupation/get-occupations");
            const formattedOccupations = response.occupations.map((occupation) => ({
                label: occupation.OccupationType,
                value: occupation._id,
            }));

            setOccupationList(formattedOccupations);
            setSelectedOccupation(formattedOccupations.length ? formattedOccupations[0] : null); // Ensure it's correctly formatted
        } catch (error) {
            setError(true);
            toast.error(error.message);
        }
    }

    // Handle Image Upload
    const handlePhotoUpload = (file, setFile, setFilePreview) => {
        if (!file) return;

        // Validate file type
        if (!file.type.startsWith("image/")) {
            toast.error("Only image files are allowed!");
            return;
        }

        // Validate file size
        if (file.size > PHOTO_MAX_FILE_SIZE) {
            toast.error(`File size exceeds the limit of (${PHOTO_MAX_FILE_SIZE / (1024 * 1024)} MB)`);
            return;
        }

        setFile(file);
        setFilePreview(URL.createObjectURL(file));
    };

    const handlePDFUpload = (file, setFile, setFilePreview) => {
        if (!file) return;

        // Validate file type
        if (!file.type.startsWith("application/pdf")) {
            toast.error("Only PDF files are allowed!");
            return;
        }

        // Validate file size
        if (file.size > PDF_MAX_FILE_SIZE) {
            toast.error(`File size exceeds the limit of (${PDF_MAX_FILE_SIZE / (1024 * 1024)} MB)`);
            return;
        }

        setFile(file);
    };

    const handleEmailChange = (value) => {
        setEmail(value);
        setCorrectEmail(isValidEmail(value));
    }

    const handleNomineeEmailChange = (value) => {
        setNomineeEmail(value);
        setCorrectNomineeEmail(isValidEmail(value));
    }

    const handleNomineePhoneNumberChange = (value) => {
        setNomineePhone(value);
        setCorrectNomineePhoneNumber(isValidPhoneNumber(value));
    }

    const handlePhoneNumberChange = (value) => {
        setPhoneNumber(value);
        setCorrectPhoneNumber(isValidPhoneNumber(value));
    }

    const handlePanNoChange = (value) => {
        setPanNo(value);
        setCorrectPanNo(isValidPanCard(value));
    }

    const handleAadharNoChange = (value) => {
        setAadharNo(value);
        setCorrectAadharNo(isValidAadharNumber(value));
    }

    const blobToFile = async (input, fileName, defaultType = "application/pdf") => {
        if (input instanceof File || input instanceof Blob) {
            return new File([input], fileName, { type: input.type || defaultType });
        } else if (typeof input === "string") {
            try {
                const response = await fetch(input);
                const blob = await response.blob();
                return new File([blob], fileName, { type: blob.type || defaultType });
            } catch (error) {
                console.error("Error converting URL to Blob:", error);
                return null; // Prevents app crash
            }
        } else {
            console.error("Invalid input type for blobToFile");
            return null;
        }
    };    
    
    const handleSubmit = async () => {
        if (!selectedBranch || !selectedTitle || !memberName || !selectedFhRelation || !fhName || !motherName || !dob || !age || !panNo || !aadharNo || !phoneNumber || !email || !selectedGender || !selectedOccupation || !monthlyIncome || !presentAddress || !companyAddress || !companyName || !nomineeAge || !nomineeEmail || !nomineeName || !nomineePhone || !nomineeRelationship || !introducersName || !photo || !signatureApplicant || !signatureIntroducer || !addressProofType || !addressProof || !aadharCard || !panCard) {
            toast("All required parameters are not filled!", { icon: "⚠️" });
            return;
        }
    
        if (sender) return;
        else setSender(true);
    
        setSubmitting(true);
    
        try {
            const formData = new FormData();
            formData.append("selectedBranch", selectedBranch.value);
            formData.append("selectedTitle", selectedTitle.value);
            formData.append("memberName", memberName);
            formData.append("selectedFhRelation", selectedFhRelation.value);
            formData.append("fhName", fhName);
            formData.append("motherName", motherName);
            formData.append("dob", dob);
            formData.append("age", age);
            formData.append("panNo", panNo);
            formData.append("aadharNo", aadharNo);
            formData.append("phoneNumber", phoneNumber);
            formData.append("email", email);
            formData.append("selectedGender", selectedGender.value);
            formData.append("selectedOccupation", selectedOccupation.value);
            formData.append("monthlyIncome", monthlyIncome);
            formData.append("presentAddress", presentAddress);
            formData.append("permanentAddress", sameAsPresent ? presentAddress : permanentAddress);
            formData.append("sameAsPresent", sameAsPresent);
            formData.append("companyAddress", companyAddress);
            formData.append("companyName", companyName);
            formData.append("nomineeAge", nomineeAge);
            formData.append("nomineeEmail", nomineeEmail);
            formData.append("nomineeName", nomineeName);
            formData.append("nomineePhone", nomineePhone);
            formData.append("introducersName", introducersName);
            formData.append("isMarried", isMarried);
            formData.append("ownHouseProperty", ownHouseProperty);
            formData.append("nomineeRelationship", nomineeRelationship);
            formData.append("addressProofType", addressProofType.value);
    
            // Append files safely
            const appendFileSafely = async (fieldName, fileInput, fileName, defaultType) => {
                const file = await blobToFile(fileInput, fileName, defaultType);
                if (file) formData.append(fieldName, file);
            };

            // Append PDF & Image files
            await appendFileSafely("addressProof", addressProof, "address_proof.pdf", "application/pdf");
            await appendFileSafely("signatureApplicant", signatureApplicant, "signature_applicant.png", "image/png");
            await appendFileSafely("signatureIntroducer", signatureIntroducer, "signature_introducer.png", "image/png");
            await appendFileSafely("aadharCard", aadharCard, "aadhar_card.pdf", "application/pdf");
            await appendFileSafely("panCard", panCard, "pan_card.pdf", "application/pdf");
            await appendFileSafely("photo", photo, "photo.png", "image/png");
    
            const response = await postAPIMedia(`/employee/update-member/${mid}`, formData);
            toast.success(response.message);
            location.reload();
        } catch (error) {
            toast.error("Error uploading data: " + error.message);
        } finally {
            setSender(false);
            setSubmitting(false);
        }
    };    

    const fetchProfileData = async () => {
        try {
            const response = await getAPI(`/employee/member/${mid}`); // Update API endpoint as needed
            const data = response.member;

            setMemberId(data.membershipNumber);
            setMemberName(data.memberName);
            setFhName(data.fhName);
            setMotherName(data.motherName);
            setDob(data.dob ? new Date(data.dob).toISOString().split("T")[0] : "");
            setAge(data.age);
            setPhoneNumber(data.phoneNumber);
            setEmail(data.email);
            setPanNo(data.panNo);
            setAadharNo(data.aadharNo);
            setMonthlyIncome(data.monthlyIncome);
            setFormattedMonthlyIncome(data.monthlyIncome);
            setPresentAddress(data.presentAddress);
            setPermanentAddress(data.permanentAddress);
            setCompanyName(data.companyName);
            setCompanyAddress(data.companyAddress);
            setNomineeName(data.nomineeName);
            setNomineeRelationship(data.nomineeRelationship);
            setNomineeAge(data.nomineeAge);
            setNomineePhone(data.nomineePhone);
            setNomineeEmail(data.nomineeEmail);
            setIntroducersName(data.introducersName);
            setSameAsPresent(data.sameAsPresent);
            setOwnHouseProperty(data.ownHouseProperty);
            setIsMarried(data.maritalStatus);

            // Set dropdown selections (Assuming the response includes IDs for selection)
            setSelectedFhRelation(data.selectedFhRelation ? fhRelation.find(t => t.label === data.selectedFhRelation) : null);
            setAddressProofType(data.addressProofType ? addressProofDocs.find(t => t.label === data.addressProofType) : null);
            setSelectedBranch(data.selectedBranch ? branches.find(t => t.value === data.selectedBranch) : null);
            setSelectedTitle(data.selectedTitle ? titles.find(t => t.value === data.selectedTitle) : null);
            setSelectedGender(data.selectedGender ? genderList.find(t => t.value === data.selectedGender) : null);
            setSelectedOccupation(data.selectedOccupation ? occupationList.find(t => t.value === data.selectedOccupation) : null);
            const fetchedPhoto = await blobFetchURL(data.photo);
            const fetchedSignatureApplicant = await blobFetchURL(data.signatureApplicant);
            const fetchedSignatureIntroducer = await blobFetchURL(data.signatureIntroducer);
            const fetchedAddressProof = await blobFetchURL(data.addressProof);
            const fetchedAdhaarCard = await blobFetchURL(data.aadharCard);
            const fetchedPanCard = await blobFetchURL(data.panCard);
            setPhoto(fetchedPhoto);
            setPhotoPreview(URL.createObjectURL(fetchedPhoto));
            setSignatureApplicant(fetchedSignatureApplicant);
            setSignatureApplicantPreview(URL.createObjectURL(fetchedSignatureApplicant));
            setSignatureIntroducer(fetchedSignatureIntroducer);
            setSignatureIntroducerPreview(URL.createObjectURL(fetchedSignatureIntroducer));
            setAddressProof(fetchedAddressProof);
            setAadharCard(fetchedAdhaarCard);
            setPanCard(fetchedPanCard);

        } catch (error) {
            setError(true);
            toast.error(error.message);
        } finally {
            setSender(false);
        }
    };

    const fetchData = async () => {
        try {
            setSender(true);

            setSelectedFhRelation(fhRelation[0]);
            setAddressProofType(addressProofDocs[0]);

            // Run all API calls in parallel
            await Promise.all([
                getBranchList(),
                getTitleList(),
                getGenderList(),
                getOccupationList(),
            ]);
        } catch (error) {
            setError(true);
            console.error(error.message);
        }
    };

    useEffect(() => {
        if (titles && genderList && branches && occupationList) fetchProfileData();
    }, [titles, genderList, branches, occupationList])

    useEffect(() => {
        fetchData();
    }, []);

    return (
        <>
            {sender ? (
                <Loader text={submitting ? "Saving..." : "Fetching Data..."} />
            ) : !error ? (
                <div className="flex overflow-y-auto px-12 py-10">
                    <div className="flex flex-col w-[70%]">
                        <div className="flex items-end justify-between">
                            <p className="text-[24px] font-semibold">{editing ? "Editing" : ""} Member Profile</p>
                            <div className="flex items-end gap-x-20">
                                <p className="flex gap-x-2 underline cursor-pointer" onClick={() => navigate(`/employee/member/${memberId}/accounts`)}>Accounts <IoOpenOutline className="text-xl" /></p>
                                {editing ? <IoCloseOutline className="text-[24px] mr-4 cursor-pointer" onClick={() => location.reload()} /> : <FaEdit onClick={() => setEditing(true)} className="text-[24px] mr-4 cursor-pointer" />}
                            </div>
                        </div>
                        <hr className="border-[#55595F] border-1 mt-5" />
                        <div className="flex flex-col mt-8 relative">
                            <label htmlFor="memberId" className="ml-1 required">Member ID</label>
                            <div className="flex gap-x-5">
                                <input
                                    maxLength="255"
                                    disabled={true}
                                    id="memberId"
                                    className="form-input bg-gray-100 mt-2 flex-1"
                                    placeholder="XXXXXXXXXX"
                                    value={memberId}
                                    required
                                />
                            </div>
                        </div>
                        <div className="flex flex-col mt-8 text-[#1E1F26]">
                            {/* Branch Selection */}
                            <div className="flex flex-col relative">
                                <label htmlFor="branchSelect" className="ml-1 required">Select Branch</label>
                                {editing ? <Select
                                    id="branchSelect"
                                    options={branches}
                                    value={selectedBranch}
                                    onChange={setSelectedBranch}
                                    className="mt-2"
                                    isSearchable
                                    placeholder="XYZ Branch"
                                    styles={{
                                        control: (base) => ({
                                            ...base,
                                            border: "1px solid #cacaca",
                                        }),
                                        container: (base) => ({
                                            ...base,
                                            width: "100%", // Make sure select takes full width
                                        }),
                                        input: (input) => ({
                                            ...input,
                                            padding: "5px 10px",
                                        }),
                                        singleValue: (singleValue) => ({
                                            ...singleValue,
                                            padding: "5px 10px"
                                        }),
                                        placeholder: (placeholder) => ({
                                            ...placeholder,
                                            padding: "5px 10px"
                                        })
                                    }}
                                /> : <input className="form-input mt-2 flex-1" value={selectedBranch?.label} disabled={true} />}
                            </div>

                            {/* Member Name */}
                            <div className="flex flex-col mt-8 relative">
                                <label htmlFor="memberName" className="ml-1 required">Member's Full Name</label>
                                <div className="flex gap-x-5">
                                    {editing ? <Select
                                        options={titles}
                                        value={selectedTitle}
                                        onChange={setSelectedTitle}
                                        className="mt-2 w-[20%]"
                                        placeholder="Title"
                                        styles={{
                                            control: (base) => ({
                                                ...base,
                                                border: "1px solid #cacaca",
                                            }),
                                            container: (base) => ({
                                                ...base,
                                                width: "20%", // Make sure select takes full width
                                            }),
                                            input: (input) => ({
                                                ...input,
                                                padding: "5px 10px",
                                            }),
                                            singleValue: (singleValue) => ({
                                                ...singleValue,
                                                padding: "5px 10px"
                                            }),
                                            placeholder: (placeholder) => ({
                                                ...placeholder,
                                                padding: "5px 10px"
                                            })
                                        }}
                                    /> : <input className="form-input max-w-[20%] mt-2 flex-1" value={selectedTitle?.label} disabled={true} />}
                                    <input
                                        maxLength="255"
                                        disabled={sender || !editing}
                                        id="memberName"
                                        className="form-input mt-2 flex-1"
                                        placeholder="John Doe"
                                        value={memberName}
                                        onChange={(e) => setMemberName(e.target.value)}
                                        required
                                    />
                                </div>
                            </div>
                            {/* Father's / Husband's Name */}
                            <div className="flex flex-col mt-8 relative">
                                <label htmlFor="fhName" className="ml-1 required">Father's / Husband's Name</label>
                                <div className="flex gap-x-5">
                                    {editing ? <Select
                                        options={fhRelation}
                                        value={selectedFhRelation}
                                        onChange={setSelectedFhRelation}
                                        className="mt-2 w-[20%]"
                                        placeholder="Relation"
                                        styles={{
                                            control: (base) => ({
                                                ...base,
                                                border: "1px solid #cacaca",
                                            }),
                                            container: (base) => ({
                                                ...base,
                                                width: "20%", // Make sure select takes full width
                                            }),
                                            input: (input) => ({
                                                ...input,
                                                padding: "5px 10px",
                                            }),
                                            singleValue: (singleValue) => ({
                                                ...singleValue,
                                                padding: "5px 10px"
                                            }),
                                            placeholder: (placeholder) => ({
                                                ...placeholder,
                                                padding: "5px 10px"
                                            })
                                        }}
                                    /> : <input className="form-input max-w-[20%] mt-2 flex-1" value={selectedFhRelation?.label} disabled={true} />}
                                    <input
                                        maxLength="255"
                                        disabled={sender || !editing}
                                        id="fhName"
                                        className="form-input mt-2 flex-1"
                                        placeholder="John Doe"
                                        value={fhName}
                                        onChange={(e) => setFhName(e.target.value)}
                                        required
                                    />
                                </div>
                            </div>
                            {/* Mother's Name */}
                            <div className="flex flex-col mt-8 relative">
                                <label htmlFor="motherName" className="ml-1 required">Mother's Name</label>
                                <div className="flex gap-x-5">
                                    <input
                                        maxLength="255"
                                        disabled={sender || !editing}
                                        id="motherName"
                                        className="form-input mt-2 flex-1"
                                        placeholder="John Doe"
                                        value={motherName}
                                        onChange={(e) => setMotherName(e.target.value)}
                                        required
                                    />
                                </div>
                            </div>
                            {/* DOB, Age, Own House Property and Marital Status */}
                            <div className="flex justify-between">
                                <div className="flex gap-x-5">
                                    <div className="flex flex-col mt-8 relative">
                                        <label htmlFor="dob" className="ml-1 required">Date of birth</label>
                                        <div className="flex gap-x-5">
                                            <input
                                                disabled={sender || !editing}
                                                type="date"
                                                id="dob"
