    /**
     * Calculates age based on the selected date of birth.
     * Prevents selecting a future date.
     * @param {Date|string} date - The selected date of birth.
     */
    const dobChange = (date) => {
        const birthDate = new Date(date);
        const today = new Date();

        if (birthDate > today) {
            toast("DOB cannot be a future date!", {
                icon: "⚠️"
            });
            return;
        }
    
        let calculatedAge = today.getFullYear() - birthDate.getFullYear();
        const monthDiff = today.getMonth() - birthDate.getMonth();
        
        // If birthday hasn't happened this year, subtract one year
        if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {
            calculatedAge--;
        }
    
        setDob(date);
        setAge(calculatedAge);
    };

            const response = await getAPI("/branchmaster/get-all");

                label: branch.branchName,

    /**
     * Fetches the membership fee structure for the currently selected branch.
     * Calculates the total and total refundable amounts based on the fetched structure.
     * Updates `membershipAmountSet`, `totalAmount`, and `totalRefundableAmount` states.
     */
    const getMembershipAmoutSet = async () => {
        try {
            const response = await getAPI(`/branchmaster/get-membership?branchId=${selectedBranch.value}`);
            const membershipSet = response.membershipAmountSet || [];
            // Calculate total amount and refundable amount
            const totalAmount = membershipSet.reduce((sum, item) => sum + Number(item.value), 0);
            const refundableAmount = membershipSet
                .filter(item => item.refundable) // Only refundable items
                .reduce((sum, item) => sum + Number(item.value), 0);

            setMembershipAmountSet(membershipSet);
            setTotalAmount(totalAmount);
            setTotalRefundableAmount(refundableAmount);
        } catch (error) {

            toast.error(error.message);

    /**

     * Validates file type (image/*) and size (PHOTO_MAX_FILE_SIZE).
     * Updates the corresponding file state and preview state.
     * @param {File} file - The uploaded file object.
     * @param {function} setFile - State setter function for the file object.
     * @param {function} setFilePreview - State setter function for the file preview URL.
     */
    const handlePhotoUpload = (file, setFile, setFilePreview) => {
        if (!file) return;

        // Validate file type
        if (!file.type.startsWith("image/")) {
            toast.error("Only image files are allowed!");
            return;
        }

        // Validate file size
        if (file.size > PHOTO_MAX_FILE_SIZE) {
            toast.error(`File size exceeds the limit of (${PHOTO_MAX_FILE_SIZE / (1024 * 1024)} MB)`);
            return;
        }

        setFile(file);
        setFilePreview(URL.createObjectURL(file));
    };

    /**
     * Handles PDF file uploads (Address Proof, Aadhar, PAN).
     * Validates file type (application/pdf) and size (PDF_MAX_FILE_SIZE).

     * @param {function} setFile - State setter function for the file object.
     * @param {function} setFilePreview - (Unused in this function) - State setter for preview.
     */

        if (!file) return;

        if (!file.type.startsWith("application/pdf")) {
            toast.error("Only PDF files are allowed!");
            return;

        // Validate file size
