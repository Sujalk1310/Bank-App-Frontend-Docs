import { useState } from "react";
import { TbUserEdit } from "react-icons/tb";
import { FaUnlockAlt } from "react-icons/fa";
import TabButton from "../../components/TabButton";
import PersonalInfo from "../../components/Admin/PersonalInfo";
import PasswordAndSecurity from '../../components/PasswordAndSecurity';

/**
 * Admin Settings page component.
 * Provides a tabbed interface for managing account settings.
 * Currently includes tabs for "Personal Info" and "Password And Security".
 * 
 * Uses `TabButton`, `PersonalInfo` (Admin variant), and `PasswordAndSecurity` (Admin variant) components.
 * 
 * @returns {JSX.Element} The admin settings page.
 */
const Settings = () => {
    const [activeTab, setActiveTab] = useState('personalInfo');

    return (
        <div className="flex flex-col w-[70%] px-12 py-10">
            <p className="text-[24px] font-semibold">Account Settings</p>
            <div className="flex gap-x-2 items-center mt-10">
                <TabButton icon={<TbUserEdit />} text={"Personal Info"} cb={() => setActiveTab('personalInfo')} activeTab={activeTab} tabName={"personalInfo"} />
                <TabButton icon={<FaUnlockAlt />} text={"Password And Security"} cb={() => setActiveTab('passwordAndSecurity')} activeTab={activeTab} tabName={"passwordAndSecurity"} />
            </div>
            <hr className="border-[#55595F] border-1" />
            {activeTab === "personalInfo" && <PersonalInfo />}
            {activeTab === "passwordAndSecurity" && <PasswordAndSecurity isAdmin={true} />}
        </div> 
    )
}

export default Settings;
