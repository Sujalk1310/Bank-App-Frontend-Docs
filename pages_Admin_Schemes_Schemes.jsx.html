import { useState, useRef } from "react";
import { LuRepeat2 } from "react-icons/lu";
import { TbPigMoney } from "react-icons/tb";
import { BiMoneyWithdraw } from "react-icons/bi";
import { HiBanknotes } from "react-icons/hi2";
import TabButton from "../../../components/TabButton";
import { IoIosArrowBack, IoIosArrowForward } from "react-icons/io";
import RD from "./RD";
import FD from "./FD";
import CD from "./CD";
import OD from "./OD";

/**
 * Schemes management page (Admin view).
 * Provides a horizontally scrollable tabbed interface to manage different deposit schemes.
 * Uses `TabButton` for scheme selection (CD, OD, RD, FD) and renders the corresponding
 * scheme management component.
 * Includes scroll buttons for navigating tabs if they overflow.
 * 
 * @returns {JSX.Element} The schemes management page.
 */
const Schemes = () => {
    const [activeTab, setActiveTab] = useState('CD');

    const scrollRef = useRef(null);

    /**
     * Scrolls the tab container horizontally.
     * @param {"left"|"right"} direction - The direction to scroll.
     */
    const scroll = (direction) => {
        if (scrollRef.current) {
            scrollRef.current.scrollBy({
                left: direction === "left" ? -150 : 150,
                behavior: "smooth",
            });
        }
    };
    
    return (
        <div className="flex-1 overflow-y-auto">
            <div className="flex flex-col w-[70%] px-12 py-10">
                <p className="text-[24px] font-semibold">Schemes</p>
                <div className="relative mt-10">
                    {/* Left Scroll Button */}
                    <button
                        className="absolute top-0 -left-[35px] z-10 h-full px-1 bg-white shadow-md border rounded-xl"
                        onClick={() => scroll("left")}
                    >
                        <IoIosArrowBack className="text-xl" />
                    </button>

                    {/* Scrollable Tab List */}
                    <div
                        ref={scrollRef}
                        className="flex gap-x-2 flex-nowrap overflow-x-auto whitespace-nowrap no-scrollbar items-center"
                    >
                        <TabButton icon={<BiMoneyWithdraw />} text={"Compulsory Deposit (CD)"} cb={() => setActiveTab('CD')} activeTab={activeTab} tabName={"CD"} />
                        <TabButton icon={<HiBanknotes />} text={"Optional Deposit (OD)"} cb={() => setActiveTab('OD')} activeTab={activeTab} tabName={"OD"} />
                        <TabButton icon={<LuRepeat2 />} text={"Recurrent Deposit (RD)"} cb={() => setActiveTab('RD')} activeTab={activeTab} tabName={"RD"} />
                        <TabButton icon={<TbPigMoney />} text={"Fixed Deposit (FD)"} cb={() => setActiveTab('FD')} activeTab={activeTab} tabName={"FD"} />
                    </div>

                    {/* Right Scroll Button */}
                    <button
                         className="absolute -right-[35px] top-0 z-10 h-full px-1 bg-white shadow-md border rounded-xl"
                        onClick={() => scroll("right")}
                    >
                        <IoIosArrowForward className="text-xl" />
                    </button>
                </div>
                <hr className="border-[#55595F] border-1 mb-4" />
                {activeTab === "CD" && <CD />}
                {activeTab === "OD" && <OD />}
                {activeTab === "RD" && <RD />}
                {activeTab === "FD" && <FD />}
            </div>
        </div>
    )
}

export default Schemes;
