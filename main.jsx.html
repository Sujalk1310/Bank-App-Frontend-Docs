/**
 * @fileoverview Main entry point for the React application.
 * Initializes the React root, sets up context providers (Google reCAPTCHA, Admin, Employee),
 * and renders the main App component.
 */
import { StrictMode } from 'react';
import { createRoot } from 'react-dom/client';
import './index.css';
import App from './App.jsx';
import { GoogleReCaptchaProvider } from 'react-google-recaptcha-v3';
import { AdminProvider } from './contexts/AdminToken';
import { EmployeeProvider } from './contexts/EmployeeToken';

/**
 * Renders the root of the React application.
 * 
 * Wraps the main App component with necessary context providers:
 * - StrictMode: For highlighting potential problems in an application.
 * - GoogleReCaptchaProvider: Provides Google reCAPTCHA functionality.
 * - AdminProvider: Provides authentication context for admin users.
 * - EmployeeProvider: Provides authentication context for employee users.
 */
createRoot(document.getElementById('root')).render(
  <StrictMode>
    <GoogleReCaptchaProvider reCaptchaKey={import.meta.env.VITE_RECAPTCHA}>
      <AdminProvider>
        <EmployeeProvider>
          <App />
        </EmployeeProvider>
      </AdminProvider>
    </GoogleReCaptchaProvider>
  </StrictMode>,
)
