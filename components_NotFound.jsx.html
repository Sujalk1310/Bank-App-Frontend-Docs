import { TbError404 } from "react-icons/tb";

/**
 * A simple component to display a "Not Found" message with a 404 icon.
 * 
 * @param {object} props - The component props.
 * @param {string} props.text - The text message to display below the icon (e.g., "Page Not Found").
 * @param {string} [props.classes] - Additional CSS classes to apply to the container div.
 * @returns {JSX.Element} The "Not Found" component.
 */
const NotFound = ({ text, classes }) => {
    return (
        <div className={classes + ' flex flex-col w-full h-full justify-center items-center'}>
            <TbError404 className="text-[100px]" />
            <p className="mt-[5px] font-medium text-[20px]">{text}</p>
        </div>
    )
}

export default NotFound;
