import React, { createContext, useState, useEffect } from "react";

/**
 * React Context for managing Employee authentication state.
 * 
 * Provides the employee token (`token`), a function to update it (`setToken`),
 * and a loading state (`loading`) indicating if the initial token check is complete.
 */
const EmployeeContext = createContext();

/**
 * Provider component for the EmployeeContext.
 * Initializes the employee token state from session storage on mount.
 * 
 * @param {object} props - The component props.
 * @param {React.ReactNode} props.children - The child components to wrap with the provider.
 * @returns {JSX.Element} The EmployeeContext provider.
 */
const EmployeeProvider = ({ children }) => {
  const [token, setToken] = useState(null);
  const [loading, setLoading] = useState(true); 

  useEffect(() => {
    const storedToken = sessionStorage.getItem("token");
    setToken(storedToken);
    setLoading(false); 
  }, []);

  return (
    <EmployeeContext.Provider value={{ token, setToken, loading }}>
      {children}
    </EmployeeContext.Provider>
  );
};

export { EmployeeContext, EmployeeProvider };
